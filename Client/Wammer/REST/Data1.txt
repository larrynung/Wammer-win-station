#region

using System.Collections.Generic;

#endregion

namespace Waveface.API.V1
{
    public class Response
    {
        public string status { get; set; }
        public string timestamp { get; set; }
        public string request_id { get; set; }
    }

    public class User
    {
        public string id { get; set; }
        public string avatar_url { get; set; }
        public string email { get; set; }
        public string nickname { get; set; }
    }

    public class Comment
    {
        public string id { get; set; }
        public string creator_id { get; set; }
        public string post_id { get; set; }
        public string creation_device_name { get; set; }
        public string text { get; set; }
        public string timestamp { get; set; }
    }

    public class File
    {
        public string id { get; set; }
        public string creator_id { get; set; }
        public string creation_device_name { get; set; }
        public string post_id { get; set; }
        public string timestamp { get; set; }
        public string url { get; set; }
        public string thumbnail_url { get; set; }
        public string orig_url { get; set; }
        public string local_name { get; set; }
        public string text { get; set; }
        public string type { get; set; }
    }

    public class OpenGraph
    {
        public string description { get; set; }
        public string provider_name { get; set; }
        public string provider_url { get; set; }
        public string thumbnail_url { get; set; }
        public string url { get; set; }
        public string title { get; set; }
        public string thumbnail_width { get; set; }
        public string thumbnail_height { get; set; }

        public string version { get; set; }
        public string type { get; set; }
    }

    public class Preview
    {
        public string id { get; set; }
        public string creator_id { get; set; }
        public string post_id { get; set; }
        public string timestamp { get; set; }
        public string type { get; set; }
        public string text { get; set; }
        public OpenGraph og { get; set; }
        public string soul { get; set; }
    }

    public class OGS_Image
    {
        public string url { get; set; }
        public string width { get; set; }
        public string height { get; set; }
    }

    public class OGS_Object
    {
        public string width { get; set; }
        public string height { get; set; }
        public string html { get; set; }
        public string type { get; set; }
    }

    public class OGS
    {
        public string provider_url { get; set; }
        //public OGS_Object @object { get; set; }
        public string description { get; set; }
        public string original_url { get; set; }
        public string url { get; set; }
        public List<OGS_Image> images { get; set; }
        public bool safe { get; set; }
        public string provider_display { get; set; }
        public string author_name { get; set; }
        public string content { get; set; }
        public string favicon_url { get; set; }
        public List<object> place { get; set; }
        public string author_url { get; set; }
        public List<object> embeds { get; set; }
        public string title { get; set; }
        public string provider_name { get; set; }
        public int cache_age { get; set; }
        public string type { get; set; }
        public List<object> @event { get; set; }
    }

    public class Preview_GP
    {
        public OpenGraph og { get; set; }
    }

    public class Preview_GAP
    {
        public OGS ogs { get; set; }
    }

    public class Post
    {
        public string id { get; set; }
        public string creator_id { get; set; }
        public string creation_device_name { get; set; }
        public string timestamp { get; set; }
        public string text { get; set; }

        public int comment_count { get; set; }
        public List<Comment> comments { get; set; }

        public int file_count { get; set; }
        public List<File> files { get; set; }

        public int preview_count { get; set; }
        public List<Preview> previews { get; set; }
    }


    public class MR_Auth_Login
    {
        public Response response { get; set; }

        public string creator_id { get; set; }
        public string token { get; set; }
    }

    public class MR_Users_FetchAll
    {
        public Response response { get; set; }                          

        public int user_count { get; set; }
        public List<User> users { get; set; }
    }

    public class MR_Posts_FetchAll
    {
        public Response response { get; set; }

        public bool is_end { get; set; }

        public int post_count { get; set; }
        public List<Post> posts { get; set; }
    }

    public class MR_Posts_FetchByConditions
    {
        public Response response { get; set; }

        public int post_count { get; set; }
        public List<Post> posts { get; set; }

        public int newer_post_count { get; set; }
        public int total_post_count { get; set; }
    }

    public class MR_Post_FetchSinglePost
    {
        public Response response { get; set; }

        public List<Post> posts { get; set; }
    }

    public class MR_Post_FetchComments
    {
        public Response response { get; set; }

        public int comment_count { get; set; }
        public List<Comment> comments { get; set; }
    }

    public class MR_Post_CreateNewPost
    {
        public Response response { get; set; }

        public Post post { get; set; }
    }

    public class MR_Post_CreateNewComment
    {
        public Response response { get; set; }

        public Comment comment { get; set; }
    }

    public class MR_File_UploadFile
    {
        public Response response { get; set; }
        public File file { get; set; }
    }

    public class MR_Preview_GetPreview
    {
        public Response response { get; set; }

        public Preview_GP preview { get; set; }
    }

    public class MR_Preview_GetAdvancedPreview
    {
        public Response response { get; set; }

        public Preview_GAP preview { get; set; }
    }

    public class MR_Preview_UploadPreview
    {
        public Response response { get; set; }

        public Preview preview { get; set; }
    }
}
